  geom_point(size = 0.8) +
  theme_void() +
  theme(legend.position = "none") +
  facet_wrap(~dataset, ncol = 3)

# Modelo por cada dataset
datasaurus_dozen %>%
  group_by(dataset) %>%
  summarize(
    intercepto  = lm(y ~ x)$coefficients[1],
    pendiente   = lm(y ~ x)$coefficients[2],
    r_cuadrado  = summary(lm(y ~ x))$r.squared
  )
